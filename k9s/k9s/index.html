Kubernetes

starting kubernetes cluster

minikube start --driver=docker

get status

minikube status

kubernetes node ip

minikube ip

check cluster info

kubectl cluster-info

get nodes

kubectl get nodes

get pods

kubectl get pods

get kubectl namespaces

kubectl get namespaces

list pods by namespaces

kubectl get pods --namespace=kube-system

run pod

kubectl run nginx --image=nginx

describe kubernetes pod

kubectl describe pod nginx

create deployment

kubectl create deployment nginx-deployment --image=nginx

descripbe deployment

kubectl describe deployment nginx-deployment

scale deployment

kubectl scale deployment nginx-deployment --replicas=2

wide pods details

kubectl get pods -o wide

create deoployment

  • kubectl create deployment my-deployment --image ragaca
  • kubectl get pods
  • kubectl expose my-deployment --port 3000
  • kubectl get services
  • kubectl delete svc my-deployment

create load balancer

  • kubectl expose deployment my-deployment --type=LoadBalancer --port 3000

update deployment

  • kubectl set image deployment my-deployment my-deployment=nginx:2.0
  • kubectl rollout status my-deployment

kubectl apply deployment

apiVersion: apps/v1
kind: Deployment
metadata:
  name: nginx-deployment
  labels:
    app: nginx
spec:
  replicas: 3
  selector:
    matchLabels:
      app: nginx
  template:
    metadata:
      labels:
        app: nginx
    spec:
      containers:
      - name: nginx
        image: nginx:1.14.2
        ports:
        - containerPort: 80

kubectl apply -f deployment.yaml

delete from file

k delete -f service.yaml -f deployment.yaml
