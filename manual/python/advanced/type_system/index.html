    ParamSpec and Variadic Types

    1

4

7
8

    from typing import Callable, TypeVar, ParamSpec

P = ParamSpec("P")
R = TypeVar("R")

def wrapper(func: Callable[P, R]) -> Callable[P, R]:
    def inner(*args: P.args, **kwargs: P.kwargs) -> R:
        return func(*args, **kwargs)
    return inner

    Type Guards

    1
2
3
4
    from typing import TypeGuard

def is_string_list(val: list[object]) -> TypeGuard[list[str]]:
    return all(isinstance(x, str) for x in val)

    from typing import overload

@overload
def process(x: int) -> str: ...

@overload
def process(x: str) -> int: ...

def process(x: int | str) -> int | str:
    if isinstance(x, int):
        return str(x)
    return int(x)

    Best Practices

    1. Type Annotations
    2. Use type hints consistently
    3. Avoid using Any when possible

    4. Prefer object over Any for type safety

    5. Generic Types

    6. Use type variables for generic code
    7. Specify bounds when appropriate

    8. Document type constraints

    9. Error Handling

    10. Use TypeGuard for runtime type checking
    11. Handle type errors gracefully

    12. Provide clear error messages
